Select 
Chk_id, 
Chk_set_id, 
Chklist.Create_DtTm, 
Chklist.Edit_DtTm, 
Chklist.Sanct_DtTm, 
Chklist.Pat_ID1, 
Chklist.Due_DtTm, 
Chklist.Elpsd_Action, 
convert(date, Chklist.Act_DtTm) as Act_Date, 
QCLTask.Description as Activity, 
Chklist.Complete,


--Rsp
Rsp_Staff_ID, Stf_Rsp.Type as Rsp_Type, Stf_Rsp.Initials as Rsp_Initial, Stf_Rsp.First_Name as Rsp_FirstName,  Stf_Rsp.Last_Name as Rsp_LastName,
--Req
Req_Staff_ID, Stf_Req.Type as Req_Type, Stf_Req.Initials as Req_Initials, Stf_Req.First_Name as Req_First,  Stf_Req.Last_Name as Req_LastName,
--Com (hold as link in Report)
Com_Staff_ID, Stf_comp.Type as Comp_Type, Stf_comp.Initials as Comp_Initials, Stf_comp.First_Name as Comp_First,  Stf_comp.Last_Name as Comp_LastName



From Chklist
LEFT OUTER JOIN QCLTask ON Chklist.TSK_ID = QCLTask.TSK_ID 
LEFT OUTER JOIN Staff Stf_comp ON Com_Staff_ID = Stf_comp.Staff_ID
LEFT OUTER JOIN Staff Stf_Rsp ON Rsp_Staff_ID = Stf_Rsp.Staff_ID
LEFT OUTER JOIN Staff Stf_Req ON Req_Staff_ID = Stf_Req.Staff_ID


Where 1=1
AND Version = 0
AND Suppressed = 0
AND NOT(Chklist.Pat_ID1 IN (&[TestPatientList]&))
AND (Due_DtTm>=CAST(  '&[ReportingStartDate]&' AS DATETIME)  
AND Due_DtTm<CAST( '&[ReportingEndDate]&' AS DATETIME))
